# Add these scripts to your package.json for easier testing

Add to "scripts" section in package.json:

"scripts": {
  "dev": "next dev -p 3001",
  "build": "next build",
  "start": "next start",
  "lint": "next lint",
  
  // NEW TESTING SCRIPTS:
  "seed": "curl -X POST http://localhost:3001/api/seed -H 'Content-Type: application/json' -d '{\"action\": \"seed\", \"useScience\": true}'",
  "test:api": "./test-api.sh",
  "test:quick": "curl http://localhost:3001/api/student-profiles?studentId=student_demo_1 | jq",
  "demo:high": "curl 'http://localhost:3001/api/assignments?role=student&mockUserId=teacher_demo_1&studentMockId=student_demo_1' | jq '.data[0] | {title, difficultyAdjustment, encouragementNote, hints, challengeQuestions}'",
  "demo:struggling": "curl 'http://localhost:3001/api/assignments?role=student&mockUserId=teacher_demo_1&studentMockId=student_demo_18' | jq '.data[0] | {title, difficultyAdjustment, encouragementNote, hints, remedialQuestions}'"
}

Then you can run:
  npm run seed           # Seed demo data
  npm run test:api       # Run full test suite
  npm run test:quick     # Quick test
  npm run demo:high      # Show high performer assignment
  npm run demo:struggling # Show struggling student assignment
